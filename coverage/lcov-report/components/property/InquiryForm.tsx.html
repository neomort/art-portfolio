import React, { useState, useEffect, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import { Calendar, MessagesSquare, Send } from 'lucide-react';
import { Button } from '../ui/button';
import { Input } from '../ui/input';
import { Card, CardHeader, CardTitle, CardContent } from '../ui/card';
import { useAuth } from '../../contexts/AuthContext';
import { supabase } from '../../lib/supabase';
import { DateRangePicker } from '../DateRangePicker';
import '../../components/DateRangePicker.css';
import { sendNotification } from '../../lib/notifications';
 
interface InquiryFormProps {
  propertyId: string;
  initialStartDate?: string;
  initialEndDate?: string;
  initialBrandInfo?: string;
}
 
const InquiryForm: React.FC<InquiryFormProps> = ({ 
  propertyId, 
  initialStartDate = '', 
  initialEndDate = '', 
  initialBrandInfo = ''
}) => {
 
  const navigate = useNavigate();
  const { user } = useAuth();
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  // Helpers for local date parsing/formatting
  function parseLocalDate(str: string | undefined): Date | null {
    if (!str) return null;
    const [year, month, day] = str.split('-').map(Number);
    if (!year || !month || !day) return null;
    return new Date(year, month - 1, day);
  }
  function formatLocalDate(date: Date | null): string {
    if (!date) return '';
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const day = String(date.getDate()).padStart(2, '0');
    return `${year}-${month}-${day}`;
  }
  // Date state as Date objects
  const [dateRange, setDateRange] = useState<[
    Date | null,
    Date | null
  ]>([
    parseLocalDate(initialStartDate),
    parseLocalDate(initialEndDate),
  ]);
  // Sync string values for form submission
  const [startDate, setStartDate] = useState<string>(initialStartDate);
  const [endDate, setEndDate] = useState<string>(initialEndDate);
  const [formData, setFormData] = useState({
    spaceRequirements: '',
    brandInfo: '',
    comments: ''
  });
 
  // Sync string values when dateRange changes
  useEffect(() => {
    setStartDate(dateRange[0] ? formatLocalDate(dateRange[0]) : '');
    setEndDate(dateRange[1] ? formatLocalDate(dateRange[1]) : '');
  }, [dateRange]);
 
  // Update local state when initial props change (e.g., from URL parameters)
  useEffect(() => {
    if (initialStartDate !== startDate) {
      setStartDate(initialStartDate);
      if (startDateRef.current) {
        startDateRef.current.value = initialStartDate;
      }
    }
  }, [initialStartDate]);
 
  useEffect(() => {
    if (initialEndDate !== endDate) {
      setEndDate(initialEndDate);
      if (endDateRef.current) {
        endDateRef.current.value = initialEndDate;
      }
    }
  }, [initialEndDate]);
 
  // Update brandInfo when initialBrandInfo changes
  useEffect(() => {
    if (initialBrandInfo && initialBrandInfo !== formData.brandInfo) {
      setFormData(prev => ({
        ...prev,
        brandInfo: initialBrandInfo
      }));
    }
  }, [initialBrandInfo]);
 
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!user) {
      navigate('/signin?redirect=' + encodeURIComponent(window.location.pathname));
      return;
    }
 
    setLoading(true);
    setError(null);
 
    try {
      // Get the ID of the newly created inquiry for idempotency
      const { data: inquiry, error: inquiryError } = await supabase
        .from('inquiries')
        .insert({
          property_id: propertyId,
          user_id: user.id,
          start_date: startDate || null,
          end_date: endDate || null,
          message: `Space Requirements:\n${formData.spaceRequirements}\n\nAbout the Brand:\n${formData.brandInfo}\n\nComments:\n${formData.comments}`,
          status: 'pending',
        })
        .select('id')
        .single();
 
      if (inquiryError) throw inquiryError;
 
      // Send notification to property owner
      try {
        // Get property details including owner info
        const { data: propertyData } = await supabase
          .from('properties')
          .select(`
            id, title,
            profiles:profiles!properties_venue_id_fkey(
              email,
              full_name
            )
          `)
          .eq('id', propertyId)
          .single();
        
        console.log('Property data for notification:', propertyData);
 
        if (propertyData?.profiles && propertyData.profiles.email) {
          console.log('Sending inquiry notification to property owner:', propertyData.profiles.email);
          console.log('Property owner details:', {
            name: propertyData.profiles.full_name || 'Property Owner',
            email: propertyData.profiles.email,
            propertyId: propertyId,
            propertyTitle: propertyData.title
          });
          
          await sendNotification(
            'new_inquiry',
            { 
              email: propertyData.profiles.email || '',
              name: propertyData.profiles.full_name || 'Property Owner'
            }, 
            { 
              requestId: `new_inquiry_${inquiry.id}`, // Use inquiry ID for idempotency
              senderName: user.full_name || 'SplitSpace User', 
              propertyTitle: propertyData.title || 'Property',
              messageContent: `Space Requirements:\n${formData.spaceRequirements}\n\nAbout the Brand:\n${formData.brandInfo}\n\nComments:\n${formData.comments}`, 
              replyLink: `${window.location.origin}/messages` 
            } 
          );
          console.log('New inquiry notification sent to property owner');
        } else {
          console.warn('Could not send notification: property owner profile or email not found', {
            hasProfiles: !!propertyData?.profiles,
            hasEmail: !!propertyData?.profiles?.email,
            propertyId
          });
        }
      } catch (notificationError) {
        console.error('Error sending notification:', notificationError);
        console.error('Notification error details:', {
          message: notificationError instanceof Error ? notificationError.message : 'Unknown error',
          propertyId,
          userId: user?.id
        });
        // Don't throw error here - we don't want to fail the inquiry if notification fails
      }
 
      // Redirect to confirmation page or dashboard
      navigate('/dashboard');
    } catch (err) {
      console.error('Error submitting inquiry:', err);
      setError(err instanceof Error ? err.message : 'Failed to submit inquiry');
    } finally {
      setLoading(false);
    }
  };
 
  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center text-[#121826]">
          <MessagesSquare className="h-5 w-5 mr-2 text-[#EA6C56]" />
          Initiate Booking Inquiry
        </CardTitle>
      </CardHeader>
      <CardContent>
        <form onSubmit={handleSubmit} className="space-y-6">
          {error && (
            <div className="bg-red-50 border border-red-200 text-red-600 px-4 py-3 rounded-xl">
              {error}
            </div>
          )}
 
          <div className="space-y-2">
            <label className="block text-sm font-medium text-[#620E28] mb-2">
              Select Dates
            </label>
            <DateRangePicker
              value={dateRange}
              onChange={setDateRange}
              minDate={new Date()}
            />
            {(initialStartDate || initialEndDate) && (
              <div className="text-sm text-green-600 bg-green-50 p-2 rounded-lg">
                ✓ Dates pre-filled from your search filters
              </div>
            )}
          </div>
 
          <div>
            <label className="block text-sm font-medium text-[#620E28] mb-2">
              Space Requirements
            </label>
            <textarea
              className="w-full rounded-xl border-2 border-maroon-200 p-3 focus:outline-none focus:ring-2 focus:ring-maroon-500 focus:border-transparent"
              rows={3}
              placeholder="Describe any specific requirements or modifications you need for the space"
              value={formData.spaceRequirements}
              onChange={(e) => setFormData({ ...formData, spaceRequirements: e.target.value })}
            />
          </div>
 
          <div>
            <label className="block text-sm font-medium text-[#620E28] mb-2">
              About my Brand
            </label>
            <textarea
              className="w-full rounded-xl border-2 border-maroon-200 p-3 focus:outline-none focus:ring-2 focus:ring-maroon-500 focus:border-transparent"
              rows={3}
              placeholder="Tell us about your brand and how you plan to use the space"
              value={formData.brandInfo}
              onChange={(e) => setFormData({ ...formData, brandInfo: e.target.value })}
            />
          </div>
 
          <div>
            <label className="block text-sm font-medium text-[#620E28] mb-2">
              Comments or Questions
            </label>
            <textarea
              className="w-full rounded-xl border-2 border-maroon-200 p-3 focus:outline-none focus:ring-2 focus:ring-maroon-500 focus:border-transparent"
              rows={3}
              placeholder="Any additional comments or questions?"
              value={formData.comments}
              onChange={(e) => setFormData({ ...formData, comments: e.target.value })}
            />
          </div>
 
          <Button
            type="submit"
            className="w-full bg-[#EA6C56] hover:bg-[#d85151] text-white"
            size="lg"
            isLoading={loading}
            icon={<Send className="h-5 w-5 mr-2" />}
          >
            Submit Inquiry
          </Button>
        </form>
      </CardContent>
    </Card>
  );
};
 
export default InquiryForm;
